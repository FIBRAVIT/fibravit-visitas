const express = require('express'); const cors = require('cors'); const bodyParser = require('body-parser'); const Database = require('better-sqlite3'); const path = require('path'); const fs = require('fs'); const app = express(); const PORT = process.env.PORT || 3000; // ── Banco de dados SQLite persistente ───────────────────────────────────── // No Render, usar /data para persistência (Persistent Disk) const DB_DIR = process.env.DB_DIR || path.join(__dirname, 'data'); const DB_PATH = path.join(DB_DIR, 'visitas.db'); if (!fs.existsSync(DB_DIR)) fs.mkdirSync(DB_DIR, { recursive: true }); const db = new Database(DB_PATH); // ── Cria tabela se não existir ───────────────────────────────────────────── db.exec(` CREATE TABLE IF NOT EXISTS visitas ( id TEXT PRIMARY KEY, data_visita TEXT, horario TEXT, empresa TEXT, coordenador TEXT, equipe TEXT, cliente TEXT, local TEXT, contrato TEXT, tipo_visita TEXT, objetivo TEXT, contato_cliente TEXT, pessoas_contatadas TEXT, proximo_contato TEXT, servicos TEXT, checkmob TEXT, fotos TEXT, materiais TEXT, recibo TEXT, pendencias TEXT, prioridade TEXT, responsavel_pend TEXT, prazo_pend TEXT, objetivos_atingidos TEXT, satisfacao TEXT, observacoes TEXT, data_envio TEXT ) `); console.log('✅ Banco de dados SQLite iniciado em:', DB_PATH); app.use(cors()); app.use(bodyParser.json({ limit: '10mb' })); app.use(express.static(path.join(__dirname, 'public'))); // ── Helpers ──────────────────────────────────────────────────────────────── function rowToObj(row) { if (!row) return null; return { 'ID': row.id, 'Data da Visita': row.data_visita, 'Horário': row.horario, 'Empresa': row.empresa, 'Coordenador': row.coordenador, 'Equipe Presente': row.equipe, 'Cliente / Evento': row.cliente, 'Local / Cidade': row.local, 'Nº Contrato / OS': row.contrato, 'Tipo de Visita': row.tipo_visita, 'Objetivo da Visita': row.objetivo, 'Houve Contato com Cliente?': row.contato_cliente, 'Pessoas Contatadas': row.pessoas_contatadas, 'Próximo Contato': row.proximo_contato, 'Serviços Executados': row.servicos, 'Registro no CheckMob?': row.checkmob, 'Fotos / Evidências?': row.fotos, 'Materiais Entregues': row.materiais, 'Recibo / NF': row.recibo, 'Itens Pendentes': row.pendencias, 'Prioridade Pendências': row.prioridade, 'Responsável Pendência': row.responsavel_pend, 'Prazo Pendência': row.prazo_pend, 'Visita Atingiu Objetivos?': row.objetivos_atingidos, 'Satisfação do Cliente': row.satisfacao, 'Observações Gerais': row.observacoes, 'Data de Envio': row.data_envio, }; } // ── ROTAS ────────────────────────────────────────────────────────────────── // Salvar nova visita app.post('/api/visita', (req, res) => { try { const d = req.body; const cnt = db.prepare('SELECT COUNT(*) as n FROM visitas').get(); const id = String(cnt.n + 1).padStart(3, '0'); const now = new Date().toISOString().split('T')[0]; db.prepare(` INSERT INTO visitas VALUES ( ?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,? ) `).run( id, d.data_visita||'', d.horario||'', d.empresa||'', d.coordenador||'', d.equipe||'', d.cliente||'', d.local||'', d.contrato||'', d.tipo_visita||'', d.objetivo||'', d.contato_cliente||'', d.pessoas_contatadas||'', d.proximo_contato||'', d.servicos||'', d.checkmob||'', d.fotos||'', d.materiais||'', d.recibo||'', d.pendencias||'', d.prioridade||'', d.responsavel_pend||'', d.prazo_pend||'', d.objetivos_atingidos||'', d.satisfacao||'', d.observacoes||'', now ); console.log(`✅ Visita ${id} salva — ${d.cliente} — ${d.coordenador}`); res.json({ success: true, id }); } catch (err) { console.error('❌ Erro:', err.message); res.status(500).json({ success: false, error: err.message }); } }); // Listar todas as visitas app.get('/api/visitas', (req, res) => { try { const rows = db.prepare('SELECT * FROM visitas ORDER BY id ASC').all(); res.json(rows.map(rowToObj)); } catch (err) { res.status(500).json({ error: err.message }); } }); // Buscar visita por ID app.get('/api/visita/:id', (req, res) => { try { const row = db.prepare('SELECT * FROM visitas WHERE id = ?').get(req.params.id); if (!row) return res.status(404).json({ error: 'Não encontrado' }); res.json(rowToObj(row)); } catch (err) { res.status(500).json({ error: err.message }); } }); // KPIs para o dashboard app.get('/api/kpis', (req, res) => { try { const data = db.prepare('SELECT * FROM visitas ORDER BY id ASC').all().map(rowToObj); const porMes={}, porTipo={}, coordenadores={}; data.forEach(v => { const mes=(v['Data da Visita']||'').substring(0,7); if(mes) porMes[mes]=(porMes[mes]||0)+1; const t=v['Tipo de Visita']||'Outros'; porTipo[t]=(porTipo[t]||0)+1; const c=v['Coordenador']||'Outros'; if(!coordenadores[c]) coordenadores[c]={total:0,comContato:0,checkmobOk:0}; coordenadores[c].total++; if(v['Houve Contato com Cliente?']==='Sim') coordenadores[c].comContato++; if(v['Registro no CheckMob?']==='Sim') coordenadores[c].checkmobOk++; }); res.json({ total: data.length, comContato: data.filter(v=>v['Houve Contato com Cliente?']==='Sim').length, semContato: data.filter(v=>v['Houve Contato com Cliente?']==='Não').length, checkmobOk: data.filter(v=>v['Registro no CheckMob?']==='Sim').length, comPendencias: data.filter(v=>v['Itens Pendentes']&&v['Itens Pendentes'].trim()).length, atingiuTotal: data.filter(v=>v['Visita Atingiu Objetivos?']==='Totalmente').length, atingiuParcial: data.filter(v=>v['Visita Atingiu Objetivos?']==='Parcialmente').length, altaPrioridade: data.filter(v=>v['Prioridade Pendências']==='Alta').length, porMes, porTipo, coordenadores, visitas: data }); } catch (err) { res.status(500).json({ error: err.message }); } }); // Health check app.get('/health', (req, res) => res.json({ status: 'ok', registros: db.prepare('SELECT COUNT(*) as n FROM visitas').get().n })); // ── Inicia ───────────────────────────────────────────────────────────────── app.listen(PORT, '0.0.0.0', () => { const n = db.prepare('SELECT COUNT(*) as n FROM visitas').get().n; console.log(''); console.log('╔══════════════════════════════════════════════╗'); console.log('║ GRUPO FIBRAVIT — Sistema de Visitas ║'); console.log(`║ Porta: ${PORT} | Registros: ${n} ║`); console.log('╚══════════════════════════════════════════════╝'); });
